\block{Signal Extraction}{
\begin{tikzfigure}[Top: invariant mass distributions; bottom: jet \pt\ distributions from signal region and side bands.\par]     \label{fig:SideBandInvMass_QM17}    \includegraphics[width=.85\colwidth]{img/SideBandInvMass_QM17} \end{tikzfigure}
\begin{itemize}
\item The \textbf{\textcolor{darkestblue}{invariant mass distributions}} of identified \Dzero-jet candidates are\\ \textbf{\textcolor{darkblue}{fitted}} in bins of \ptd\ to extract the peak position and width and the\\ background normalization $B'$
\item The \ptchjet\ distributions of the \textbf{\textcolor{lightgreen}{side bands}} are subtracted from those of the \textbf{\textcolor{lightred}{signal region}} and weighted by the efficiency $\epsilon_{\Dzero}(\ptd)$ % (Fig.~\ref{fig:HQ16_Simulation_EfficiencyVsDP})
\end{itemize}
\centering
 $\textcolor{darkerblue}{N(\ptchjet)}=\sum_{\ptd}\frac{1}{\epsilon_{\Dzero}(\ptd)} \cdot \left[\textcolor{lightred}{N_{\rm Sign}(\ptd,\ptchjet)}-B'\textcolor{lightgreen}{N_{\rm SB}(\ptd,\ptchjet)} \right]$
}

\block{Conclusions and Outlook}{
\begin{minipage}[t]{0.40\colwidth}
\begin{tikzfigure}[Relative uncertainties\par]     \label{fig:Uncertainties_QM17}    \includegraphics[width=\textwidth]{img/Uncertainties_QM17} \end{tikzfigure}
\end{minipage}
\begin{minipage}[t]{0.55\colwidth}
\begin{itemize}
\item ALICE has measured the cross section of charm jets tagged with fully reconstructed \Dzero\ mesons in the range $5<\ptchjet<30$~\GeVc\ in \pp\ collisions at $\s=7$~TeV
\item POWHEG interfaced with PYTHIA6 (Perugia-2011) is in agreement with the measurement
\item The precision of the measurement is limited by statistics
\item In the near future we will extend this measurement to the larger datasets at $\s=8,13$~TeV using electromagnetic calorimeters for triggering and full jet reconstruction
\item The goal is the measurement of the fragmentation of charm jets in an extended kinematic region
\end{itemize}
\end{minipage}
